    <script>
        // Copy to clipboard function
        function copyToClipboard(text) {
            navigator.clipboard.writeText(text).then(function() {
                showNotification('Copié dans le presse-papier', 'success');
            }, function(err) {
                showNotification('Échec de la copie', 'error');
            });
        }
        
        // Show notification
        function showNotification(message, type) {
            const container = document.getElementById('notification-container');
            const notification = document.createElement('div');
            
            notification.className = `notification px-4 py-3 rounded-md shadow-md text-white ${
                type === 'success' ? 'bg-green-500' : 'bg-red-500'
            }`;
            notification.innerHTML = `
                <div class="flex items-center">
                    <i class="fas ${type === 'success' ? 'fa-check-circle' : 'fa-exclamation-circle'} mr-2"></i>
                    <span>${message}</span>
                </div>
            `;
            
            container.appendChild(notification);
            
            setTimeout(() => {
                notification.style.opacity = '0';
                setTimeout(() => {
                    container.removeChild(notification);
                }, 300);
            }, 3000);
        }
        
        // Modal functions
        function openModal() {
            document.getElementById('deposit-modal').classList.remove('hidden');
        }
        
        function closeModal() {
            document.getElementById('deposit-modal').classList.add('hidden');
        }
        
        // Payment method toggle
        document.getElementById('payment-method').addEventListener('change', function() {
            const trc20 = document.getElementById('trc20-instructions');
            const mobileMoney = document.getElementById('mobile-money-instructions');
            
            if (this.value === 'trc20') {
                trc20.classList.remove('hidden');
                mobileMoney.classList.add('hidden');
            } else {
                trc20.classList.add('hidden');
                mobileMoney.classList.remove('hidden');
            }
        });
        
        // Show TRC20 address when clicked
        function showTrc20Address() {
            const addressInput = document.getElementById('trc20-address');
            addressInput.classList.remove('hidden-address');
            copyToClipboard('TCoXMtAuPgo2t44XXasFjpATFCNXxwktgK');
        }
        
        // Initialize payment method display
        document.getElementById('trc20-instructions').classList.remove('hidden');
        
        // Make blurred addresses visible on click
        document.querySelectorAll('.hidden-address').forEach(input => {
            input.addEventListener('click', function() {
                this.classList.remove('hidden-address');
            });
        });
